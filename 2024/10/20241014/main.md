vector<bool> is_prime(n + 1, true);
vector<int> primes;
is_prime[0] = is_prime[1] = false;

for (ll i = 2; i <= n; i++) {
  if (is_prime[i]) {
    primes.emplace_back(i);
  }
  for (ll j = 0; j < primes.size() && i * primes[j] <= n; j++) {
    is_prime[i * primes[j]] = false;
    if (i % primes[j] == 0) {
      break;
    }
  }
}
